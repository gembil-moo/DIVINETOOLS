# DIVINETOOLS
Tools utility untuk manajemen paket dan optimasi di Android (Termux).

# 1. Install paket yang dibutuhkan
pkg install curl unzip

# 2. Download dan ekstrak proyek
curl -L -o DIVINETOOLS.zip "https://github.com/gembil-moo/DIVINETOOLS/archive/refs/heads/main.zip"
unzip DIVINETOOLS.zip

# 3. Masuk ke folder dan jalankan installer
cd DIVINETOOLS-main/
bash install.sh
